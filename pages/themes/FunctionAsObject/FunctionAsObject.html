The Function object

In JavaScript every function is actually an instance of Function built-in object.
The global Function object has no methods or properties of its own, but each function inherits some properties and methods through the prototype chain from Function.prototype

Examples


            // function declaration:
            function foo(x,y) {
                console.log(`x:${x}, y: ${y}`);
                console.log(`this: ${this}`);
            }

            // function properties:
            console.log(`foo.name: ${foo.name}`);// name of the function
            console.log(`foo.length: ${foo.length}`); // arguments count
        


            function foo(x,y) {
                console.log(`x:${x}, y: ${y}`);
                console.log(`this: ${this}`);
            }
            // normal function call (note the 'this' value):
            console.log(`Normal foo()call:`);
            foo(2,3);

            // function methods (note the 'this' value):
            console.log(`Calling foo() with apply():`);
            foo.apply({id:1},[2,3]); // arguments are passed as array

            console.log(`Calling foo() with call():`);
            foo.call({id:1},2,3);   // arguments are passed as comma separated list

            console.log(`Binding foo:`);
            let new_foo = foo.bind({id:1},20,30);   // arguments are passed as comma separated list,and they will be prepended to the new_foo() arguments, if given
            new_foo(8,9);

Task: fix the bug

In next example, a User constructor is defined.
The greet() methods works as expected, when called immediately, but when we pass it to setTimeout(), we don't have the expected output.
You're task is to fix the bug.

            function User(userName) {
                this.userName = userName
              }
              User.prototype.greet = function () {
                console.log(`Hi, I'm ${this.userName}`);
              }

              const pesho = new User('Peter');

              // Calling greet method immediately: works ok
              pesho.greet();
              // Hi, I'm Peter

              // Calling greet method after 2sec: bug!
              setTimeout(pesho.greet,2000);
              // Hi, I'm undefined
